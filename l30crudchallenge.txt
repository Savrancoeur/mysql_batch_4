=> CRUD Challenge

CREATE DATABASE IF NOT EXISTS shirtsdb;

SHOW DATABASES;

USE shirtsdb;

CREATE TABLE IF NOT EXISTS shirts(
    id INT NOT NULL AUTO_INCREMENT,
    article VARCHAR(100),
    color VARCHAR(100),
    size VARCHAR(100),
    PRIMARY KEY (id)
);

DESC shirts;
SHOW CREATE TABLE shirts;
